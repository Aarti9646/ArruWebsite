2. Create a simple AWT application that displays a button with Custom Layout.

Code


import java.awt.*;

public class CustomLayoutAWTApplication {
    public static void main(String[] args) {
        // Create the frame
        Frame frame = new Frame("Custom Layout AWT Application");

        // Create a button
        Button button = new Button("Click Me!");

        // Set the layout manager to custom layout
        frame.setLayout(new CustomLayout());

        // Add the button to the frame
        frame.add(button);

        // Set the frame size
        frame.setSize(300, 200);

        // Center the frame on the screen
        Dimension screenSize = Toolkit.getDefaultToolkit().getScreenSize();
        int x = (screenSize.width - frame.getWidth()) / 2;
        int y = (screenSize.height - frame.getHeight()) / 2;
        frame.setLocation(x, y);

        // Make the frame visible
        frame.setVisible(true);

        // Add a window listener to handle closing the frame
        frame.addWindowListener(new java.awt.event.WindowAdapter() {
            public void windowClosing(java.awt.event.WindowEvent windowEvent) {
                System.exit(0);
            }
        });
    }
}

class CustomLayout implements LayoutManager {
    public void addLayoutComponent(String name, Component comp) {
        // Not used
    }

    public void layoutContainer(Container parent) {
        Component[] components = parent.getComponents();
        int x = (parent.getWidth() - components[0].getWidth()) / 2;
        int y = (parent.getHeight() - components[0].getHeight()) / 2;
        components[0].setBounds(x, y, components[0].getWidth(), components[0].getHeight());
    }

    public Dimension minimumLayoutSize(Container parent) {
        return new Dimension(0, 0);
    }

    public Dimension preferredLayoutSize(Container parent) {
        return new Dimension(200, 100);
    }

    public void removeLayoutComponent(Component comp) {
        // Not used
    }
}
